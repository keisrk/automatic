\item[{Procedure}] \leavevmode 
(Boudet, Comon)

\end{description}

 \begin{definition}[Automaton algorithm]
 Construct the automaton \( \mathcal{A}_\varphi \) serving as the interpreter of \( \varphi \), .i.e, \( \mathcal{L}(\mathcal{A}_\varphi) \) contains all the solutions of \( \varphi \).
 \begin{itemize}
 \item \( \mathcal{A}_{\varphi} \) accepts the solution of an atomic formula \( \varphi \).
 \item \( \mathcal{A}_{\varphi_1 \wedge \varphi_2} \) accepts \( \mathcal{L}(\mathcal{A}_{\varphi_1}) \cap \mathcal{L}(\mathcal{A}_{\varphi_2}) \).
 \item \( \mathcal{A}_{\varphi_1 \vee \varphi_2} \) accepts \( \mathcal{L}(\mathcal{A}_{\varphi_1}) \cup \mathcal{L}(\mathcal{A}_{\varphi_2}) \).
 \item \( \mathcal{A}_{\neg \varphi} \) accepts the complement of \( \mathcal{L}(\mathcal{A}_{\varphi}) \).
 \item \( \mathcal{A}_{\exists x_i .\ \varphi} \) accepts the \( i \)-th projection of \( \mathcal{L}(\mathcal{A}_{\varphi}) \).
 \end{definition}

#. Regular set as solution space
#. Check language emptiness

\begin{theorem}[Boudet, Comon]
    The automaton algorithm decides Presburger arithmetic in triple
    exponential time.
\end{theorem}
